
if(PKG_CONFIG_FOUND)
  pkg_search_module(LIBOPENCV "opencv >= 2.4.9.1")
endif()

if(NOT LIBOPENCV_FOUND)
  find_package(OpenCV CONFIG QUIET)
  if(OpenCV_FOUND)
    set(LIBOPENCV_FOUND TRUE)
    set(LIBOPENCV_INCLUDE_DIRS ${OpenCV_INCLUDE_DIRS})
    set(LIBOPENCV_LIBRARIES ${OpenCV_LIBS})
  endif()
endif()

if(NOT LIBOPENCV_FOUND)
  find_path(LIBOPENCV_INCLUDE_DIR opencv2/opencv.hpp)
  find_library(LIBOPENCV_LIBRARY NAMES opencv_core)
  if(LIBOPENCV_INCLUDE_DIR AND LIBOPENCV_LIBRARY)
    set(LIBOPENCV_FOUND TRUE)
    set(LIBOPENCV_INCLUDE_DIRS ${LIBOPENCV_INCLUDE_DIR})
    set(LIBOPENCV_LIBRARIES ${LIBOPENCV_LIBRARY})
  endif()
endif()

if(LIBOPENCV_FOUND)
  set(CONF_DO_BUILD TRUE)
  set(CONF_HEADER_DIRS
    "${LIB_BASE}/header_files")
  set(CONF_INCLUDE_DIRS
    ${LIBOPENCV_INCLUDE_DIRS})
  set(CONF_COMP_DEFS)
  set(CONF_LINK_LIBS
    base_ucll
    ${LIBOPENCV_LIBRARIES})
endif()
