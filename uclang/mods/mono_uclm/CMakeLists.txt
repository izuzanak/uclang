
if(PKG_CONFIG_FOUND)
  pkg_search_module(LIBGLIB2 "glib-2.0")
  pkg_search_module(LIBMONOSGEN "monosgen-2 >= 3.2.8")
endif()

if(NOT LIBGLIB2_FOUND)
  find_package(GLib2 CONFIG QUIET)
  if(GLib2_FOUND)
    set(LIBGLIB2_FOUND TRUE)
    if(TARGET GLib2::glib)
      get_target_property(LIBGLIB2_INCLUDE_DIRS GLib2::glib
          INTERFACE_INCLUDE_DIRECTORIES)
      set(LIBGLIB2_LIBRARIES GLib2::glib)
    endif()
  endif()
endif()

if(NOT LIBGLIB2_FOUND)
  find_path(LIBGLIB2_INCLUDE_DIR glib.h PATH_SUFFIXES glib-2.0)
  find_library(LIBGLIB2_LIBRARY NAMES glib-2.0)
  if(LIBGLIB2_INCLUDE_DIR AND LIBGLIB2_LIBRARY)
    set(LIBGLIB2_FOUND TRUE)
    set(LIBGLIB2_INCLUDE_DIRS ${LIBGLIB2_INCLUDE_DIR})
    set(LIBGLIB2_LIBRARIES ${LIBGLIB2_LIBRARY})
  endif()
endif()

if(NOT LIBMONOSGEN_FOUND)
  find_package(Mono CONFIG QUIET)
  if(Mono_FOUND)
    set(LIBMONOSGEN_FOUND TRUE)
    if(TARGET Mono::monosgen)
      get_target_property(LIBMONOSGEN_INCLUDE_DIRS Mono::monosgen
          INTERFACE_INCLUDE_DIRECTORIES)
      set(LIBMONOSGEN_LIBRARIES Mono::monosgen)
    endif()
  endif()
endif()

if(NOT LIBMONOSGEN_FOUND)
  find_path(LIBMONOSGEN_INCLUDE_DIR mono/jit/jit.h PATH_SUFFIXES mono-2.0)
  find_library(LIBMONOSGEN_LIBRARY NAMES monosgen-2.0)
  if(LIBMONOSGEN_INCLUDE_DIR AND LIBMONOSGEN_LIBRARY)
    set(LIBMONOSGEN_FOUND TRUE)
    set(LIBMONOSGEN_INCLUDE_DIRS ${LIBMONOSGEN_INCLUDE_DIR})
    set(LIBMONOSGEN_LIBRARIES ${LIBMONOSGEN_LIBRARY})
  endif()
endif()

if(LIBGLIB2_FOUND AND LIBMONOSGEN_FOUND)
  set(CONF_DO_BUILD TRUE)
  set(CONF_HEADER_DIRS
    "${LIB_BASE}/header_files")
  set(CONF_INCLUDE_DIRS
    ${LIBGLIB2_INCLUDE_DIRS}
    ${LIBMONOSGEN_INCLUDE_DIRS})
  set(CONF_COMP_DEFS)
  set(CONF_LINK_LIBS
    base_ucll
    ${LIBGLIB2_LIBRARIES}
    ${LIBMONOSGEN_LIBRARIES})
endif()
