
if(PKG_CONFIG_FOUND)
  pkg_search_module(LIBSQLITE3 "sqlite3 >= 3.8.7.1")
endif()

if(NOT LIBSQLITE3_FOUND)
  find_package(SQLite3 CONFIG QUIET)
  if(SQLite3_FOUND)
    set(LIBSQLITE3_FOUND TRUE)
    if(TARGET SQLite::SQLite3)
      get_target_property(LIBSQLITE3_INCLUDE_DIRS SQLite::SQLite3
          INTERFACE_INCLUDE_DIRECTORIES)
      set(LIBSQLITE3_LIBRARIES SQLite::SQLite3)
    endif()
  endif()
endif()

if(NOT LIBSQLITE3_FOUND)
  find_path(LIBSQLITE3_INCLUDE_DIR sqlite3.h)
  find_library(LIBSQLITE3_LIBRARY NAMES sqlite3)
  if(LIBSQLITE3_INCLUDE_DIR AND LIBSQLITE3_LIBRARY)
    set(LIBSQLITE3_FOUND TRUE)
    set(LIBSQLITE3_INCLUDE_DIRS ${LIBSQLITE3_INCLUDE_DIR})
    set(LIBSQLITE3_LIBRARIES ${LIBSQLITE3_LIBRARY})
  endif()
endif()

if(LIBSQLITE3_FOUND)
  set(CONF_DO_BUILD TRUE)
  set(CONF_HEADER_DIRS
    "${LIB_BASE}/header_files")
  set(CONF_INCLUDE_DIRS
    ${LIBSQLITE3_INCLUDE_DIRS})
  set(CONF_COMP_DEFS)
  set(CONF_LINK_LIBS
    base_ucll
    ${LIBSQLITE3_LIBRARIES})
endif()
