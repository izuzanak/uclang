
if(PKG_CONFIG_FOUND)
  pkg_search_module(LIBMPG123 "libmpg123 >= 1.20.1")
endif()

if(NOT LIBMPG123_FOUND)
  find_package(mpg123 CONFIG QUIET)
  if(mpg123_FOUND)
    set(LIBMPG123_FOUND TRUE)
    if(TARGET MPG123::libmpg123)
      get_target_property(LIBMPG123_INCLUDE_DIRS MPG123::libmpg123
          INTERFACE_INCLUDE_DIRECTORIES)
      set(LIBMPG123_LIBRARIES MPG123::libmpg123)
    endif()
  endif()
endif()

if(NOT LIBMPG123_FOUND)
  find_path(LIBMPG123_INCLUDE_DIR mpg123.h)
  find_library(LIBMPG123_LIBRARY NAMES mpg123)
  if(LIBMPG123_INCLUDE_DIR AND LIBMPG123_LIBRARY)
    set(LIBMPG123_FOUND TRUE)
    set(LIBMPG123_INCLUDE_DIRS ${LIBMPG123_INCLUDE_DIR})
    set(LIBMPG123_LIBRARIES ${LIBMPG123_LIBRARY})
  endif()
endif()

if(LIBMPG123_FOUND)
  set(CONF_DO_BUILD TRUE)
  set(CONF_HEADER_DIRS
    "${LIB_BASE}/header_files")
  set(CONF_INCLUDE_DIRS
    ${LIBMPG123_INCLUDE_DIRS})
  set(CONF_COMP_DEFS)
  set(CONF_LINK_LIBS
    base_ucll
    ${LIBMPG123_LIBRARIES})
endif()
